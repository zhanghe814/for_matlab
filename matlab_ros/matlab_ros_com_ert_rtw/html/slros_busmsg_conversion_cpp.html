164// Conversions between SL_Bus_matlab_ros_com_std_msgs_Header and std_msgs::Header



165



166void convertFromBus(std_msgs::Header* msgPtr, SL_Bus_matlab_ros_com_std_msgs_Header const* busPtr)



167{



168  const std::string rosMessageType("std_msgs/Header");



169



170  convertFromBusVariablePrimitiveArray(msgPtr->frame_id, busPtr->FrameId, busPtr->FrameId_SL_Info);



171  msgPtr->seq =  busPtr->Seq;



172  convertFromBus(&msgPtr->stamp, &busPtr->Stamp);



173}



174



175void convertToBus(SL_Bus_matlab_ros_com_std_msgs_Header* busPtr, std_msgs::Header const* msgPtr)



176{



177  const std::string rosMessageType("std_msgs/Header");



178



179  convertToBusVariablePrimitiveArray(busPtr->FrameId, busPtr->FrameId_SL_Info, msgPtr->frame_id, slros::EnabledWarning(rosMessageType, "frame_id"));



180  busPtr->Seq =  msgPtr->seq;



181  convertToBus(&busPtr->Stamp, &msgPtr->stamp);



182}
